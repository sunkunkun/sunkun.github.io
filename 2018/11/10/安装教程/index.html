#网易云音乐修复
sudo gedit /etc/sudoers
修改/etc/sudoers文件，加一行： YOURNAME ALL = NOPASSWD: /usr/bin/netease-cloud-music
YOURNAME为你登录的用户名
sudo gedit /usr/share/applications/netease-cloud-music.desktop
修改Exec=netease-cloud-music %U 为 Exec=sudo netease-cloud-music %U, 这样点击网易云音乐图标就是以管理员权限启动的了，且不用输入密码了。
于此同时设置root/admin等没有密码登录

#终端美化
sudo apt install zsh
sudo apt install git
sh -c “$(wget https://raw.github.com/robbyrussell/oh-my-zsh/master/tools/install.sh -O -)”
chsh -s /bin/zsh
echo $SHELL #查看当前shell
cat /etc/shells #查看系统已安装的shell
git clone https://github.com/zsh-users/zsh-autosuggestions /home/.oh-my-zsh/plugins/zsh-autosuggestions
git clone https://github.com/zsh-users/zsh-syntax-highlighting.git /home/.oh-my-zsh/plugins/home/.oh-my-zsh/plugins/zsh-syntax-highlighting
sudo -i
cd /home/.oh-my-zsh/plugins
mkdir incr
git clone http://mimosa-pudica.net/src/incr-0.2.zsh /home/.oh-my-zsh/plugins/incr
cat /home/.zshrc
ZSH_THEME=”ys”
ENABLE_CORRECTION=”true”
COMPLETION_WAITING_DOTS=”true”
plugins=( git autojump incr zsh-autosuggestions zsh-syntax-highlighting )

status=/tmp/keyboard_led_status
if [[ ${1} == “on” ]] || [[ ! -e ${status} ]]; then
xset led named ‘Scroll Lock’ && echo “on” > ${status}
else
xset -led named ‘Scroll Lock’ && rm -f ${status}
fi
